控制的灯泡按键代码
控制一个灯亮灭
if(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_8)==GPIO_PIN_SET)//判断是否有按键按下
{
delay(500); //延时消抖
if(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_8)==GPIO_PIN_SET)//如果确实有按键按下
{
HAL_GPIO_Togglepin(GPIOB,GPIO_PIN_13);//对B13端口进行电频翻转
while(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_8)==GPIO_PIN_SET);//等待按键按下松开
}
}

控制两盏灯亮灭
if(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_9)==GPIO_PIN_SET)//判断是否有按键按下
{
delay(500); //延时消抖
if(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_9)==GPIO_PIN_SET)//如果确实有按键按下
{
HAL_GPIO_Write(GPIOB,GPIO_PIN_13|GPIO_PIN_14,GPIO_PIN_RESET);//把两个端口输出低电平
while(HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_9)==GPIO_PIN_SET);//等待按键按下松开
}
}
